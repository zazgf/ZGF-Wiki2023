anaconda

安装docker

sudo apt-get remove docker docker-ce docker-io 
sudo apt-get install debootstrap
sudo apt install docker.io
systemctl restart docker

构建docker

ln -sf /usr/local/cuda-10.0/targets/aarch64-linux/lib/libcusparse.so /usr/lib/libcusparse.so.10
docker pull arm64v8/ubuntu:18.04
apt-get update
apt-get install python3 python3-pip
pip3 install --upgrade pip
apt-get install libhdf5-serial-dev hdf5-tools libhdf5-dev zlib1g-dev zip libjpeg8-dev liblapack-dev libblas-dev gfortran pkg-config
ln -sf /usr/lib/aarch64-linux-gnu/hdf5/serial/libhdf5.so /usr/lib/libhdf5.so

pip install -U numpy==1.16.1
pip install -U h5py==2.10.0

pip install --pre --extra-index-url https://developer.download.nvidia.com/compute/redist/jp/v45 tensorflow==2.3.1 -i http://mirrors.aliyun.com/pypi/simple/ --trusted-host mirrors.aliyun.com
ln -sf /usr/local/cuda-10.0/targets/aarch64-linux/lib/libcudart.so /usr/lib/libcudart.so.10.2
ln -sf /usr/lib/aarch64-linux-gnu/libcuda.so /usr/lib/libcuda.so.1
ln -sf /usr/lib/aarch64-linux-gnu/tegra/lib* /usr/lib/
ln -sf /usr/local/cuda-10.0/targets/aarch64-linux/lib/libcublas.so /usr/lib/libcublas.so.10
ln -sf /usr/local/cuda-10.0/targets/aarch64-linux/lib/libcusparse.so /usr/lib/libcusparse.so.10
ln -sf /usr/local/cuda-10.0/targets/aarch64-linux/lib/libcusolver.so /usr/lib/libcusolver.so.10
ln -sf /usr/local/cuda-10.0/targets/aarch64-linux/lib/libcurand.so /usr/lib/libcurand.so.10
ln -sf /usr/local/cuda-10.0/targets/aarch64-linux/lib/libcufft.so /usr/lib/libcufft.so.10

AGX sounddevice

// 可以使用 aplay 来测试, 哪个出声音就是哪个了 ...
aplay -D hw:0,3 /usr/share/sounds/alsa/test.wav
aplay -D hw:0,7 /usr/share/sounds/alsa/test.wav
aplay -D hw:0,8 /usr/share/sounds/alsa/test.wav
